Google Sheets -synkronointi
Tulokset tallentuvat automaattisesti Google Sheetsiin jokaisen pelin jälkeen. Tarvitset Apps Script -webhookin — näytä ohjeet ▾
  1. Avaa Google Sheets → Laajennukset → Apps Script
  2. Korvaa kaikki koodi alla olevalla ja tallenna (Ctrl+S)
  3. Klikkaa Ota käyttöön → Uusi käyttöönotto
  4. Tyyppi: Verkkosovellus · Suorita: Minä · Käyttöoikeus: Kaikki, myös anonyymit
  5. Kopioi Verkkosovelluksen URL kenttään alta
// ── Apps Script — Darts kaksisuuntainen synkronointi ── // Kopioi tämä koodi Google Apps Scriptiin ja ota käyttöön verkkosovelluksena. var SS = SpreadsheetApp.getActiveSpreadsheet(); // ── doPost: vastaanottaa datan sovellukselta ── function doPost(e) { var data = JSON.parse(e.postData.contents); var action = data.action || 'append'; if (action === 'append' && data.sheet && data.rows) { var sh = SS.getSheetByName(data.sheet) || SS.insertSheet(data.sheet); if (data.header && sh.getLastRow() === 0) sh.appendRow(data.header); data.rows.forEach(function(row) { sh.appendRow(row); }); } if (action === 'syncPlayers' && data.players) { var sh = SS.getSheetByName('Pelaajat') || SS.insertSheet('Pelaajat'); sh.clearContents(); sh.appendRow(['nimi','emoji','pelejä','voittoja','pisteitä','tikkoja','kierroksia','busteja','tuplat','triplat','paras','paras_kierros_heitot','paras_kierros_pisteet','päivitetty']); data.players.forEach(function(p) { var br = p.bestRound; sh.appendRow([p.name, p.emoji||'🎯', p.games||0, p.wins||0, p.totalScore||0, p.totalDarts||0, p.totalRounds||0, p.busts||0, p.doubles||0, p.triples||0, p.bestGame||'', br?br.throws.join('|'):'', br?br.score:'', new Date().toISOString()]); }); } if (action === 'syncDiary' && data.entries) { var sh = SS.getSheetByName('Päiväkirja') || SS.insertSheet('Päiväkirja'); sh.clearContents(); sh.appendRow(['ts','mode','pelaajat_json']); data.entries.forEach(function(e) { sh.appendRow([e.ts, e.mode||'', JSON.stringify(e.results)]); }); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } // ── doGet: palauttaa kaiken datan sovellukselle ── function doGet(e) { var result = { ok: true, players: [], diary: [] }; var pSh = SS.getSheetByName('Pelaajat'); if (pSh && pSh.getLastRow() > 1) { var rows = pSh.getRange(2, 1, pSh.getLastRow()-1, 14).getValues(); result.players = rows.map(function(r) { var brThrows = r[11] ? String(r[11]).split('|') : null; var brScore = r[12] ? Number(r[12]) : null; return { name: r[0], emoji: r[1]||'🎯', games: Number(r[2])||0, wins: Number(r[3])||0, totalScore: Number(r[4])||0, totalDarts: Number(r[5])||0, totalRounds: Number(r[6])||0, busts: Number(r[7])||0, doubles: Number(r[8])||0, triples: Number(r[9])||0, bestGame: r[10]!==''?Number(r[10]):null, bestRound: brThrows&&brScore?{throws:brThrows,score:brScore}:null }; }); } var dSh = SS.getSheetByName('Päiväkirja'); if (dSh && dSh.getLastRow() > 1) { var dRows = dSh.getRange(2, 1, dSh.getLastRow()-1, 3).getValues(); result.diary = dRows.map(function(r) { try { return {ts: Number(r[0]), mode: r[1], results: JSON.parse(r[2])}; } catch(e) { return null; } }).filter(Boolean); } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
